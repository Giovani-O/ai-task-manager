# Tasks API Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create two REST API routes for tasks: a paginated list route and a single task fetch route

- [ ] **Step 1: Create the route file with GET /tasks endpoint**

```typescript
import { desc, sql, type SQL } from 'drizzle-orm'
import { createSelectSchema } from 'drizzle-zod'
import type { FastifyPluginAsyncZod } from 'fastify-type-provider-zod'
import { z } from 'zod'
import { tasks, users } from '@/db/schema'

const SORT_COLUMNS = {
  title: tasks.title,
  estimatedTime: tasks.estimatedTime,
  createdAt: tasks.createdAt,
} as const

type SortColumn = keyof typeof SORT_COLUMNS

export const listTasks: FastifyPluginAsyncZod = async (app) => {
  app.get(
    '/tasks',
    {
      schema: {
        summary: 'List tasks with pagination and sorting',
        tags: ['task'],
        querystring: z.object({
          page: z.coerce.number().min(1).default(1),
          pageSize: z.coerce.number().min(1).max(100).default(20),
          sortBy: z.enum(['title', 'estimatedTime', 'createdAt']).default('createdAt'),
          sortDirection: z.enum(['asc', 'desc']).default('desc'),
        }),
        response: {
          200: z.object({
            tasks: z.array(
              z.object({
                id: z.string(),
                title: z.string(),
                estimatedTime: z.string(),
                createdAt: z.date(),
                userName: z.string(),
              }),
            ),
            page: z.number(),
            pageSize: z.number(),
          }),
        },
      },
    },
    async (request, reply) => {
      const { page, pageSize, sortBy, sortDirection } = request.query
      const offset = (page - 1) * pageSize

      const sortColumn = SORT_COLUMNS[sortBy]
      const order = sortDirection === 'asc' ? sortColumn : desc(sortColumn)

      const result = await app.db
        .select({
          id: tasks.id,
          title: tasks.title,
          estimatedTime: tasks.estimatedTime,
          createdAt: tasks.createdAt,
          userName: users.name,
        })
        .from(tasks)
        .innerJoin(users, tasks.authorId, users.id)
        .orderBy(order)
        .limit(pageSize)
        .offset(offset)

      return reply.status(200).send({
        tasks: result,
        page,
        pageSize,
      })
    },
  )
}
```

### Task 3: Create Get Task By ID Route

**Files:**

- [ ] **Step 1: Create the get-task route file**

```typescript
import { eq } from 'drizzle-orm'
import { createSelectSchema } from 'drizzle-zod'
import type { FastifyPluginAsyncZod } from 'fastify-type-provider-zod'
import { z } from 'zod'
import { tasks, users } from '@/db/schema'

export const getTask: FastifyPluginAsyncZod = async (app) => {
  app.get(
    '/tasks/:id',
    {
      schema: {
        summary: 'Get a single task by ID',
        tags: ['task'],
        params: z.object({
          id: z.string().uuid(),
        }),
        response: {
          200: z.object({
            task: createSelectSchema(tasks).and(
              z.object({
                userName: z.string(),
              }),
            ),
          }),
          404: z.object({
            error: z.string(),
          }),
        },
      },
    },
    async (request, reply) => {
      const { id } = request.params

      const result = await app.db
        .select({
          id: tasks.id,
          authorId: tasks.authorId,
          title: tasks.title,
          description: tasks.description,
          steps: tasks.steps,
          estimatedTime: tasks.estimatedTime,
          implementationSuggestion: tasks.implementationSuggestion,
          acceptanceCriteria: tasks.acceptanceCriteria,
          suggestedTests: tasks.suggestedTests,
          content: tasks.content,
          chatHistory: tasks.chatHistory,
          createdAt: tasks.createdAt,
          updatedAt: tasks.updatedAt,
          userName: users.name,
        })
        .from(tasks)
        .innerJoin(users, tasks.authorId, users.id)
        .where(eq(tasks.id, id))
        .limit(1)

      if (result.length === 0) {
        return reply.status(404).send({ error: 'Task not found' })
      }

      return reply.status(200).send({ task: result[0] })
    },
  )
}
```
